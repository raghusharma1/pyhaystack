23class HaystackSession(object): 

24 """ 

25 The Haystack Session handler is responsible for presenting an API for 

26 querying and controlling a Project Haystack server. 

27 

28 HaystackSession itself is the base class, which is then implemented by way 

29 of HaystackOperation subclasses which are instantiated by the session 

30 object before being started and returned. 

31 

32 These operations by default are specified by class member references 

33 to the classes concerned. 

34 

35 Methods for Haystack operations return an 'Operation' object, which 

36 may be used in any of two ways: 

37 

38 - as a synchronous result placeholder by calling its `wait` method 

39 followed by inspection of its `result` attribute. 

40 - as an asynchronous call manager by connecting a "slot" (`callable` 

41 that takes keyword arguments) to the `done_sig` signal. 

42 

43 The base class takes some arguments that control the default behaviour of 

44 the object. 

45 """

123 def authenticate(self, callback=None): 

124 """ 

125 Authenticate with the Project Haystack server. If an authentication 

126 attempt is in progress, we return it, otherwise we instantiate a new 

127 one. 

128 """ 

129 if self._auth_op is not None: 

130 auth_op = self._auth_op() 

131 else: 

132 auth_op = None 

133 

134 new = auth_op is None 

135 if new: 

136 auth_op = self._AUTH_OPERATION(self) 

137 auth_op.done_sig.connect(self._on_authenticate_done) 

138 

139 if callback is not None: 

140 if auth_op.is_done: 

141 # Already done 

142 return callback(auth_op) 

143 else: 

144 auth_op.done_sig.connect(callback) 

145 

146 if new: 

147 auth_op.go() 

148 self._auth_op = weakref.ref(auth_op) 

149 

150 return auth_op

454 def _on_read(self, ids, filter_expr, limit, callback, **kwargs): 

455 if isinstance(ids, string_types) or isinstance(ids, hszinc.Ref): 

456 # Make sure we always pass a list. 

457 ids = [ids] 

458 

459 if bool(ids): 

460 if filter_expr is not None: 

461 raise ValueError("Either specify ids or filter_expr, not both") 

462 

463 ids = [self._obj_to_ref(r) for r in ids] 

464 

465 if len(ids) == 1: 

466 # Reading a single entity 

467 return self._get_grid("read", callback, args={"id": ids[0]}, **kwargs) 

468 else: 

469 # Reading several entities 

470 grid = hszinc.Grid() 

471 grid.column["id"] = {} 

472 grid.extend([{"id": r} for r in ids]) 

473 return self._post_grid("read", grid, callback, **kwargs) 

474 else: 

475 args = {"filter": filter_expr} 

476 if limit is not None: 

477 args["limit"] = int(limit) 

478 

479 return self._get_grid("read", callback, args=args, **kwargs)
